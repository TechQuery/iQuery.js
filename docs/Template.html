// 模板求值

    $.Template(
        "[ ${this.time} ]  Hello, ${scope.creator}'s ${view.name} !",
        ['view', 'scope']
    ).evaluate(
        {time: '2015-04-30'},
        {name: 'iQuery.js'},
        {creator: 'TechQuery'}
    )
